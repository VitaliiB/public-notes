```js

  return new Promise((resolve, reject) => {
    try {
      отправитьЗапрос()
        .then(result => {
          if (result.success) {
            resolve(result.data);
          } else {
            reject({ message: 'Ошибка' });
          }
        })
        .catch(e => reject(e));
    } catch (e) {
      reject(e);
    }
  });
}
```
